90 uint32_t generate_crc32c(uint8_t* buf, size_t buflen)
91 {
92  unsigned int i;
93  uint32_t crc32 = ~0L;
94  uint32_t result = 0;
95  uint8_t byte0 = 0, byte1 = 0, byte2 = 0, byte3 = 0;
96 
97  for (i = 0; i < buflen; i++){
98  CRC32C(crc32, buf[i]);
99  }
100  result = ~crc32;
101 
102  /* result now holds the negated polynomial remainder;
103  * since the table and algorithm is "reflected" [williams95].
104  * That is, result has the same value as if we mapped the message
105  * to a polynomial, computed the host-bit-order polynomial
106  * remainder, performed final negation, then did an end-for-end
107  * bit-reversal.
108  * Note that a 32-bit bit-reversal is identical to four inplace
109  * 8-bit reversals followed by an end-for-end byteswap.
110  * In other words, the bytes of each bit are in the right order,
111  * but the bytes have been byteswapped. So we now do an explicit
112  * byteswap. On a little-endian machine, this byteswap and
113  * the final ntohl cancel out and could be elided.
114  */
115 
116  byte0 = (result >> 0) & 0xff;
117  byte1 = (result >> 8) & 0xff;
118  byte2 = (result >> 16) & 0xff;
119  byte3 = (result >> 24) & 0xff;
120 
121  crc32 = ((byte0 << 24) | (byte1 << 16) | (byte2 << 8) | (byte3 << 0));
122  return ( crc32 );
123 }
