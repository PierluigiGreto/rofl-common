580  /* Components of a OXM TLV header. */

582  (((CLASS) << 16) | ((FIELD) << 9) | ((HASMASK) << 8) | (LENGTH))

584  OXM_HEADER__(CLASS, FIELD, 0, LENGTH)

589  #define OXM_TYPE(HEADER) (((HEADER) >> 9) & 0x7fffff)
590  #define OXM_HASMASK(HEADER) (((HEADER) >> 8) & 1)
591  #define OXM_LENGTH(HEADER) ((HEADER) & 0xff)
