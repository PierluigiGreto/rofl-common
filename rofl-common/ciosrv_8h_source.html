286  cioloop(pthread_t tid = 0) :
287  tid(tid),
288  keep_on_running(false) {
289 
290  if (0 == tid) {
291  this->tid = pthread_self();

294  struct rlimit rlim;

299  RwLock wlock(wfds_rwlock, RwLock::RWLOCK_WRITE);

301  rfds.push_back(NULL);
302  wfds.push_back(NULL);
