428 class fppp_lcp_option : public fframe {
429 public:
430  struct fpppframe::ppp_lcp_opt_hdr_t *hdr;
431 public:
432  fppp_lcp_option(uint8_t *opt, size_t optlen) :
433  fframe(opt, optlen), hdr((struct fpppframe::ppp_lcp_opt_hdr_t*)opt) {};
434  fppp_lcp_option(fpppframe::ppp_lcp_opt_hdr_t *opt, size_t optlen) :
435  fframe((uint8_t*)opt, optlen), hdr(opt) {};
436  void validate(uint16_t total_len = 0) const {
437  if (hdr->length < sizeof(uint16_t)) throw ePPPLcpOptionInvalid();
438  switch (hdr->option) {
439  case fpppframe::PPP_LCP_OPT_RESERVED: {
440  if (hdr->length < 2) throw ePPPLcpOptionInvalid();
441  }
442  break;
443  case fpppframe::PPP_LCP_OPT_MRU: {
444  if (hdr->length < 4) throw ePPPLcpOptionInvalid();
445  }
446  break;
447  case fpppframe::PPP_LCP_OPT_ACCM: {
448  if (hdr->length < 6) throw ePPPLcpOptionInvalid();
449  }
450  break;
451  case fpppframe::PPP_LCP_OPT_AUTH_PROT: {
452  if (hdr->length < 4) throw ePPPLcpOptionInvalid();
453  }
454  break;
455  case fpppframe::PPP_LCP_OPT_QUAL_PROT: {
456  if (hdr->length < 4) throw ePPPLcpOptionInvalid();
457  }
458  break;
459  case fpppframe::PPP_LCP_OPT_MAGIC_NUM: {
460  if (hdr->length < 6) throw ePPPLcpOptionInvalid();
461  }
462  break;
463  case fpppframe::PPP_LCP_OPT_PFC: {
464  if (hdr->length < 2) throw ePPPLcpOptionInvalid();
465  }
466  break;
467  case fpppframe::PPP_LCP_OPT_ACFC: {
468  if (hdr->length < 2) throw ePPPLcpOptionInvalid();
469  }
470  break;
471  default: {
472  // do nothing
473  }
474  break;
475  }
476  };
477  uint16_t
478  get_mru() throw (ePPPLcpOptionInvalid) {
479  if (fpppframe::PPP_LCP_OPT_MRU != hdr->option) throw ePPPLcpOptionInvalid();
480  return (uint16_t)((hdr->data[0] << 8) + hdr->data[1]);
481  };
482  uint32_t
483  get_accm() throw (ePPPLcpOptionInvalid) {
484  if (fpppframe::PPP_LCP_OPT_ACCM != hdr->option) throw ePPPLcpOptionInvalid();
485  return (uint32_t)((hdr->data[0] << 24) + (hdr->data[1] << 16) + (hdr->data[2] << 8) + (hdr->data[3] << 0));
486  };
487  uint32_t
488  get_magic_num() throw (ePPPLcpOptionInvalid) {
489  if (fpppframe::PPP_LCP_OPT_MAGIC_NUM != hdr->option) throw ePPPLcpOptionInvalid();
490  return (uint32_t)((hdr->data[0] << 24) + (hdr->data[1] << 16) + (hdr->data[2] << 8) + (hdr->data[3] << 0));
