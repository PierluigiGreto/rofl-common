261  cjrnentry&
262  add_log_entry() {
263  AcquireReadWriteLock rwlock(entries_rwlock);
264  while (entries.size() > max_entries) {
265  uint32_t entry_id = entries.begin()->first;
266  delete entries[entry_id];
267  entries.erase(entry_id);
268  }
269  while (entries.find(last_entry_id) != entries.end()) {
270  last_entry_id++;
271  }
272  entries[last_entry_id] = new cjrnentry(last_entry_id);
273  return *(entries[last_entry_id]);
274  };
